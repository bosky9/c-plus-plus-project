18  template <typename T, bool Destruct>
19  struct ObjectStorage
20  {
21  using TStorage = typename std::aligned_storage<sizeof(T), std::alignment_of<T>::value>::type;
22 
23  ObjectStorage() : data() {}
24 
25  ObjectStorage(const ObjectStorage& other)
26  {
27  new(&data) T(other.stored_object());
28  }
29 
30  ObjectStorage(ObjectStorage&& other)
31  {
32  new(&data) T(std::move(other.stored_object()));
33  }
34 
35  ~ObjectStorage() { destruct_on_exit<T>(); }

37  template <typename... Args>
38  void construct(Args&&... args)
39  {
40  new (&data) T(std::forward<Args>(args)...);
41  }

43  template <bool AllowManualDestruction = !Destruct>
44  typename std::enable_if<AllowManualDestruction>::type destruct()
45  {
46  stored_object().~T();
47  }
48 
49  private:
50  // If this is a constructor benchmark, destruct the underlying object
51  template <typename U>
52  void destruct_on_exit(typename std::enable_if<Destruct, U>::type* = 0) { destruct<true>(); }
53  // Otherwise, don't
54  template <typename U>
55  void destruct_on_exit(typename std::enable_if<!Destruct, U>::type* = 0) { }
56 
57  T& stored_object() {
58  return *static_cast<T*>(static_cast<void*>(&data));
59  }
60 
61  T const& stored_object() const {
