15 // This is a simple implementation of C++11 Uniform Random Number
16 // Generator. It does not provide all operators, because Catch2
17 // does not use it, but it should behave as expected inside stdlib's
18 // distributions.
19 // The implementation is based on the PCG family (http://pcg-random.org)
20 class SimplePcg32 {
21 using state_type = std::uint64_t;
22 public:
23 using result_type = std::uint32_t;
24 static constexpr result_type (min)() {
25 return 0;
26 }
27 static constexpr result_type (max)() {
28 return static_cast<result_type>(-1);
29 }
30
31 // Provide some default initial state for the default constructor
32 SimplePcg32():SimplePcg32(0xed743cc4U) {}
33
34 explicit SimplePcg32(result_type seed_);
35
36 void seed(result_type seed_);
37 void discard(uint64_t skip);
38
39 result_type operator()();
40
41 private:
42 friend bool operator==(SimplePcg32 const& lhs, SimplePcg32 const& rhs);
43 friend bool operator!=(SimplePcg32 const& lhs, SimplePcg32 const& rhs);
44
45 // In theory we also need operator<< and operator>>
46 // In practice we do not use them, so we will skip them for now
47
48
49 std::uint64_t m_state;
50 // This part of the state determines which "stream" of the numbers
51 // is chosen -- we take it as a constant for Catch2, so we only
52 // need to deal with seeding the main state.
53 // Picked by reading 8 bytes from `/dev/random` :-)
54 static const std::uint64_t s_inc = (0x13ed0cc53f939476ULL << 1ULL) | 1ULL;
55 };
